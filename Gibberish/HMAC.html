def self.digest(key, data, opts={})
  digest_type = opts[:digest] || :sha1
  if opts[:binary]
    OpenSSL::HMAC.digest(DIGEST[digest_type], key, data)
  else
    OpenSSL::HMAC.hexdigest(DIGEST[digest_type], key, data)
  end
end
